# 一、安装
## 1、MySQL8.x
```bash
Mac M1：
docker run --name mysql \
--platform=linux/amd64 \
-p 3306:3306 \
-e MYSQL_ROOT_HOST='%' \
-e MYSQL_ROOT_PASSWORD=123456 \
-d amd64/mysql:5.7.36

Mac M3：
docker run --name mysql \
-p 3306:3306 \
-e MYSQL_ROOT_HOST='%' \
-e MYSQL_ROOT_PASSWORD=123456 \
-d mysql:8.0.32
```

```bash
[root@localhost ~]# docker pull mysql
[root@localhost ~]# mkdir -p /docker/mysql/conf
[root@localhost ~]# vim /docker/mysql/conf/my.cnf    # 设置 MySQL 字符编码
```

```bash
[client]
default-character-set=utf8mb4

[mysql]
default-character-set=utf8mb4

[mysqld]
port=3306
max_connections=200
default-time_zone='+8:00'
# password 加密方式，8.0 之前默认使用 mysql_native_password，8.0 默认使用 caching_sha2_password
default_authentication_plugin= mysql_native_password
character-set-client-handshake=0

character-set-server=utf8mb4
# ci(case insensitive)：MySQL 对大小写不敏感
# utf8mb4_unicode_ci：支持德法俄等特殊字符
# utf8mb4_general_ci：不支持，但速度比 utf8utf8mb4_unicode_ci 稍快
collation-server=utf8mb4_unicode_ci

init_connect='SET NAMES utf8mb4 COLLATE utf8mb4_unicode_ci' 

# 禁止 MySQL 对客户端连接进行 DNS 解析，可以省去域名解析的时间，使用后，客户端必须用 ip:port 连接
skip-name-resolve

# Linux 下表名是区分大小写的，1：不区分
lower_case_table_names=1


#事务隔离级别，默认为可重复读，mysql默认可重复读级别（此级别下可能参数很多间隙锁，影响性能）
transaction_isolation = READ-COMMITTED
#在MySQL暂时停止响应新请求之前的短时间内多少个请求可以被存在堆栈中
#官方建议 back_log = 50 + (max_connections / 5),封顶数为65535,默认值= max_connections
back_log = 110
# 慢 sql 相关：
#是否启用慢查询日志，1为启用，0为禁用  
slow_query_log = 1
#记录系统时间
log_timestamps = SYSTEM
#指定慢查询日志文件的路径和名字
slow_query_log_file = /var/lib/mysql/slow.log
#慢查询执行的秒数，必须达到此值可被记录
long_query_time = 5
# 查询行数大于 5000 的慢sql才会被记录
min_examined_row_limit = 5000
#作为从库时生效, 从库复制中如何有慢sql也将被记录
#对于ROW格式binlog，不管执行时间有没有超过阈值，都不会写入到从库的慢查询日志
log_slow_slave_statements = 1
```

```bash

-v /docker/mysql/log:/var/log/mysql \
-v /docker/mysql/data:/var/lib/mysql \
-v /docker/mysql/conf:/etc/mysql \
-v /docker/mysql/mysql-files:/var/lib/mysql-files \
--restart=always --privileged \
-e MYSQL_ROOT_PASSWORD=123456 -d mysql
# 说明： MySQL8 必须挂载 /var/lib/mysql-files/ 目录，不然启动报错
```

## 2、MySQL8.x Cluster
> Master、Slave 中一个重启，主从复制就断了，要重新设置主从复制！！

### 2.1、一主一从
> Master：192.168.1.2:3316
>
> Slave：192.168.1.2:3326

**1、配置 Master**

```bash
[root@localhost ~]# mkdir -p /docker/mysql/cluster/master/conf
[root@localhost ~]# vi /docker/mysql/cluster/master/conf/my.cnf
```

```bash
# 1、基本配置
[client]
default-character-set=utf8mb4

[mysql]
default-character-set=utf8mb4

[mysqld]
port=3306
max_connections=200
default-time_zone='+8:00'
default_authentication_plugin= mysql_native_password
character-set-client-handshake=0
character-set-server=utf8mb4
collation-server=utf8mb4_unicode_ci
init_connect='SET NAMES utf8mb4 COLLATE utf8mb4_unicode_ci' 
skip-name-resolve
lower_case_table_names=1

# 2、主从复制
# Master ID，要保证唯一
server_id=1
# binlog 文件名
log-bin=mysql-bin
# 主机读写
read-only=0
# 以下数据库要备份
binlog-do-db=mall_order
# 以下数据库不备份
replicate-ignore-db=mysql
replicate-ignore-db=sys
replicate-ignore-db=information_schema
replicate-ignore-db=performance_schema
# binlog 格式
binlog_format=MIXED
```

```bash
docker run -p 3316:3306 --name mysql-master \
-v /docker/mysql/cluster/master/log:/var/log/mysql \
-v /docker/mysql/cluster/master/data:/var/lib/mysql \
-v /docker/mysql/cluster/master/conf:/etc/mysql \
-v /docker/mysql/cluster/master/mysql-files:/var/lib/mysql-files \
--restart=always --privileged \
-e MYSQL_ROOT_PASSWORD=123456 -d mysql

[root@localhost ~]# docker ps -a
[root@localhost ~]# docker exec -it mysql-master bash
root@27420a08059d:/# mysql -uroot -p

# 为 Master 创建账户密码并授予复制权限，允许 Slave 通过这个账号密码来访问，实际工作中 % 可能要换成 Slave 的 IP？
mysql> CREATE USER 'slave'@'%' IDENTIFIED BY '123456';
mysql> GRANT REPLICATION SLAVE ON *.* TO 'slave'@'%';
mysql> ALTER USER 'slave'@'%' IDENTIFIED WITH mysql_native_password BY '123456';
mysql> FLUSH PRIVILEGES;
mysql> SHOW MASTER STATUS;  # 记下 File 和 Position 的值，此时不要对 Master 进行任何写操作，不然这两项的值会变！
```

```bash
docker run -p 3326:3306 --name mysql-slave \
-v /docker/mysql/cluster/slave/log:/var/log/mysql \
-v /docker/mysql/cluster/slave/data:/var/lib/mysql \
-v /docker/mysql/cluster/slave/conf:/etc/mysql \
-v /docker/mysql/cluster/slave/mysql-files:/var/lib/mysql-files \
--restart=always --privileged \
-e MYSQL_ROOT_PASSWORD=123456 -d mysql

[root@localhost ~]# docker ps -a
[root@localhost ~]# docker exec -it mysql-slave bash
root@27420a08059d:/# mysql -uroot -p

# Slave 使用 Master 授权的账号密码访问 Master
mysql> CHANGE MASTER TO MASTER_HOST='192.168.1.2', MASTER_USER='slave', MASTER_PASSWORD='123456', MASTER_LOG_FILE='mysql-bin.000003', MASTER_LOG_POS=1111, MASTER_PORT=3316;
mysql> START SLAVE;				# 启动主从同步
mysql> SHOW SLAVE STATUS\G;		# 查看 Slave 状态，若出现如下状态，说明配置成功，此时就可以主从复制了
# Slave_IO_Runing: YES
# Slave_SQL_Runing: YES
```

```bash
mysql> STOP SLAVE;      # 停止 I/O 线程和 SQL 线程的操作
mysql> RESET SLAVE;     # 在从机上执行，删除 slave 的 relaylog 文件，并重新启用新的 relaylog 文件
mysql> RESET MASTER;    # 在主机上执行，删除 master 的 binglog 文件，并重新开始所有新的日志文件
```

> 在 master 中创建数据库 mall_order，再创建表并插入数据，在 slave 中查看是否成功复制！

### 2.2、双主双从
> - Master1：192.168.1.2:3316
> - Slave1：192.168.1.2:3326
> - Master2：192.168.1.2:3336
> - Slave2：192.168.1.2:3346

**1、配置 Master1**

```bash
[root@localhost ~]# mkdir -p /docker/mysql/cluster/master1/conf
[root@localhost ~]# vi /docker/mysql/cluster/master1/conf/my.cnf
```

```bash
# 1、基本配置
[client]
default-character-set=utf8mb4

[mysql]
default-character-set=utf8mb4

[mysqld]
port=3306
max_connections=200
default-time_zone='+8:00'
default_authentication_plugin= mysql_native_password
character-set-client-handshake=0
character-set-server=utf8mb4
collation-server=utf8mb4_unicode_ci
init_connect='SET NAMES utf8mb4 COLLATE utf8mb4_unicode_ci' 
skip-name-resolve
lower_case_table_names=1

# 2、主从复制
# Master ID
server_id=1
# binlog 文件名
log-bin=mysql-bin
# 主机读写
read-only=0
# 以下数据库要备份
binlog-do-db=mall_order
# 以下数据库不备份
replicate-ignore-db=mysql
replicate-ignore-db=sys
replicate-ignore-db=information_schema
replicate-ignore-db=performance_schema
# binlog 格式
binlog_format=MIXED

# master1、master2 互为主备，所以 master2 必须复制 master1 的数据，
# 当 mater1 宕机时，master2 上位，所以当 master1 恢复时，也要复制 master2 的数据，因此 master1、master2 要相互复制！
# 问题：直接写数据库，则该操作会被写入 binlog，但通过主从复制写数据库，则该操作并不会写入 binlog！
# 如：直接写 mster2，则写操作会被写入 master2 的 binlog，
# 但直接写 master1，master2 复制 master1，并不会写入 master2 的 binlog！
# 开启该参数，使主从复制的操作也会被写入 binlog！
log-slave-updates
# 自增 id 的初值 ∈[1, 65535]；自增步长 ∈[1, 65535]，默认为 1
# 因为要配置双主双从，双主都可以写，所以 master1 初值设为 1， master2 初值设为 2，两者自增步长都设为 2
# 这样两者的 id 就不会重复了！
auto-increment-offset=1
auto-increment-increment=2
```
